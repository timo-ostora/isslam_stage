<?php

namespace App\Filament\Resources\ModuleResource\RelationManagers;

use App\Models\Assessment;
use App\Models\Lesson;
use Filament\Resources\RelationManagers\RelationManager;
use Filament\Tables;
use Filament\Tables\Table;
use Filament\Schemas\Schema;
use Filament\Schemas\Components\Section;
use Filament\Schemas\Components\Select; // Unified v5 Form/Infolist components
use Filament\Schemas\Get; // Unified state retriever for v5

class ModuleItemsRelationManager extends RelationManager
{
    protected static string $relationship = 'moduleItems';
    protected static ?string $title = 'Module Items';


    public function schema(Schema $schema): Schema
    {
        return $schema->components([
            Section::make('Content Type')
                ->columns(2)
                ->schema([
                    Select::make('itemable_type')
                        ->label('Content Type')
                        ->options([
                            Lesson::class     => 'Lesson',
                            Assessment::class => 'Assessment',
                        ])
                        ->required()
                        ->live()
                        ->native(false),

                    Select::make('itemable_id')
                        ->label('Select Content')
                        ->options(fn (Get $get) => match ($get('itemable_type')) {
                            Lesson::class     => Lesson::query()->pluck('title', 'id'),
                            Assessment::class => Assessment::query()->pluck('title', 'id'),
                            default           => [],
                        })
                        ->searchable()
                        ->required()
                        ->disabled(fn (Get $get) => blank($get('itemable_type')))
                        ->native(false),
                ]),
        ]);
    }

    public function table(Table $table): Table
    {
        return $table
            ->recordTitleAttribute('itemable_id')
            ->reorderable('position')
            ->defaultSort('position')
            ->columns([
                Tables\Columns\TextColumn::make('position')
                    ->label('#'),

                Tables\Columns\TextColumn::make('content_type')
                    ->label('Type')
                    ->badge()
                    ->color(fn (string $state) => $state === 'lesson' ? 'info' : 'warning'),

                Tables\Columns\TextColumn::make('itemable.title')
                    ->label('Title')
                    ->weight('bold'),
            ])
            ->headerActions([
                Tables\Actions\CreateAction::make(),
            ])
            ->actions([
                Tables\Actions\EditAction::make(),
                Tables\Actions\DeleteAction::make(),
            ]);
    }
}
